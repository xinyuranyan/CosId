# CosId-MyBatis 模块

cosid-mybatis (opens new window) 拦截MyBatis插入(Insert)请求,并解析 @CosId 注入分布式ID。

# 安装

Kotlin DSL

    implementation("me.ahoo.cosid:cosid-mybatis:${cosidVersion}")

public class Order {

    @CosId(value = "order")
    private Long orderId;
    private Long userId;

    public Long getOrderId() {
        return orderId;
    }

    public void setOrderId(Long orderId) {
        this.orderId = orderId;
    }

    public Long getUserId() {
        return userId;
    }

    public void setUserId(Long userId) {
        this.userId = userId;
    }
}

@Mapper
public interface OrderRepository {
    @Insert("insert into t_table (id) value (#{id});")
    void insert(LongIdEntity order);

                    "</foreach>",
            "</script>"})
    void insertList(List<FriendlyIdEntity> list);
}
